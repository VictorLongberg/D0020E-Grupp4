class Lecture {

	}

		var stud = new Student(id, name);
		this.students.push(stud);
	}

	get_student_name(id){
		for (var i = 0; i < this.students.length; i++){
			if (this.students[i].session_id == id){
				return this.students[i].name;
			}
		}
		return "Unknown";
	}
}
